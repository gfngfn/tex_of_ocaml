\documentclass{jsarticle}
\usepackage{secd}
\makeatletter
\tracingmacros=1\relax
%% Evaluate `(fun x -> "foo" ^ "bar" ^ x) "baz"`
\edef\resultA{%
  \secd@EvalAndShowString{%
    \secd@Run{}{}{%
      \secd@CLOSURE{%
        \secd@CONST{\secd@STRING{foo}\secd@ENDVAL}%
        \secd@CONST{\secd@STRING{bar}\secd@ENDVAL}%
        \secd@PRIM{\secd@PrimitiveStringConcat}%
        \secd@ACCESS{}%
        \secd@PRIM{\secd@PrimitiveStringConcat}%
      }%
      \secd@CONST{\secd@STRING{baz}\secd@ENDVAL}%
      \secd@APPLY{}%
    }{}%
  }%
}
\edef\resultB{%
  \secd@EvalAndShowInt{%
    \secd@Run{}{}{%
      \secd@CONST{\secd@INT{oooo}\secd@ENDVAL}%
    }{}%
  }%
}
\tracingmacros=0\relax
\makeatother
\begin{document}
  \resultA, \resultB
\end{document}
